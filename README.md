import re
from pathlib import Path
from datetime import datetime, timedelta

def search_logs(log_dir='data/logs', search_term=None, level=None, hours=24):
    """Search logs with filters"""
    
    log_dir = Path(log_dir)
    cutoff = datetime.now() - timedelta(hours=hours)
    results = []
    
    for log_file in log_dir.rglob('*.log'):
        with open(log_file, 'r') as f:
            for line in f:
                # Parse timestamp (adjust format as needed)
                match = re.match(r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})', line)
                if match:
                    log_time = datetime.strptime(match.group(1), '%Y-%m-%d %H:%M:%S')
                    if log_time < cutoff:
                        continue
                
                # Filter by level
                if level and level not in line:
                    continue
                
                # Filter by search term
                if search_term and search_term.lower() not in line.lower():
                    continue
                
                results.append(line.strip())
    
    return results

# Usage
errors = search_logs(level='ERROR', hours=24)
detections = search_logs(search_term='detection', hours=1)

def analyze_error_rate(log_file='data/logs/system/system.log', hours=24):
    """Analyze error rates over time"""
    
    from collections import defaultdict
    from datetime import datetime
    
    error_counts = defaultdict(int)
    cutoff = datetime.now() - timedelta(hours=hours)
    
    with open(log_file, 'r') as f:
        for line in f:
            if 'ERROR' in line:
                # Extract hour
                match = re.match(r'(\d{4}-\d{2}-\d{2} \d{2}):', line)
                if match:
                    hour = match.group(1)
                    error_counts[hour] += 1
    
    total_errors = sum(error_counts.values())
    avg_per_hour = total_errors / hours if hours > 0 else 0
    
    return {
        'total_errors': total_errors,
        'avg_per_hour': avg_per_hour,
        'hourly': dict(error_counts),
        'peak_hour': max(error_counts, key=error_counts.get) if error_counts else None
    }

def analyze_performance(log_file='data/logs/performance/latency.log'):
    """Analyze performance metrics from logs"""
    
    latencies = []
    
    with open(log_file, 'r') as f:
        for line in f:
            # Extract latency values
            match = re.search(r'latency[=:](\d+(?:\.\d+)?)', line, re.IGNORECASE)
            if match:
                latencies.append(float(match.group(1)))
    
    if latencies:
        return {
            'count': len(latencies),
            'min_ms': min(latencies),
            'max_ms': max(latencies),
            'avg_ms': sum(latencies) / len(latencies),
            'p50': sorted(latencies)[len(latencies)//2],
            'p95': sorted(latencies)[int(len(latencies)*0.95)],
            'p99': sorted(latencies)[int(len(latencies)*0.99)]
        }
    
    return None

import gzip
import shutil
from pathlib import Path
from datetime import datetime, timedelta

def cleanup_old_logs(log_dir='data/logs', days=30, compress=True):
    """Clean up old log files"""
    
    log_dir = Path(log_dir)
    cutoff = datetime.now() - timedelta(days=days)
    cleaned = []
    
    for log_file in log_dir.rglob('*.log'):
        # Skip if not old enough
        mtime = datetime.fromtimestamp(log_file.stat().st_mtime)
        if mtime > cutoff:
            continue
        
        if compress and log_file.stat().st_size > 1024 * 1024:  # > 1MB
            # Compress old log
            with open(log_file, 'rb') as f_in:
                with gzip.open(f"{log_file}.gz", 'wb') as f_out:
                    shutil.copyfileobj(f_in, f_out)
            
            # Remove original
            log_file.unlink()
            cleaned.append(f"{log_file}.gz")
        else:
            # Delete old log
            log_file.unlink()
            cleaned.append(str(log_file))
    
    return cleaned

# Schedule daily cleanup
def schedule_cleanup():
    import schedule
    schedule.every().day.at("02:00").do(cleanup_old_logs)

def aggregate_logs(log_dir='data/logs', output_file='logs_aggregated.json'):
    """Aggregate logs for analysis"""
    
    import json
    
    aggregated = {
        'system': [],
        'api': [],
        'hardware': [],
        'security': [],
        'performance': []
    }
    
    for category in aggregated.keys():
        category_dir = Path(log_dir) / category
        if category_dir.exists():
            for log_file in category_dir.glob('*.log'):
                with open(log_file, 'r') as f:
                    aggregated[category].extend(f.readlines())
    
    with open(output_file, 'w') as f:
        json.dump(aggregated, f, indent=2)
    
    return aggregated

import logging
import logging.handlers
from pathlib import Path

def setup_logging(log_level='INFO', log_dir='data/logs'):
    """Configure logging for the application"""
    
    log_dir = Path(log_dir)
    log_dir.mkdir(parents=True, exist_ok=True)
    
    # Create formatters
    detailed_formatter = logging.Formatter(
        '%(asctime)s [%(levelname)-8s] [%(name)s] [%(filename)s:%(lineno)d] %(message)s',
        datefmt='%Y-%m-%d %H:%M:%S'
    )
    
    json_formatter = logging.Formatter(
        '{"timestamp": "%(asctime)s", "level": "%(levelname)s", "logger": "%(name)s", "message": "%(message)s"}'
    )
    
    # System log handler
    system_handler = logging.handlers.RotatingFileHandler(
        log_dir / 'system/system.log',
        maxBytes=100 * 1024 * 1024,  # 100 MB
        backupCount=30
    )
    system_handler.setFormatter(detailed_formatter)
    system_handler.setLevel(logging.INFO)
    
    # Error log handler
    error_handler = logging.handlers.RotatingFileHandler(
        log_dir / 'system/error.log',
        maxBytes=50 * 1024 * 1024,  # 50 MB
        backupCount=10
    )
    error_handler.setFormatter(detailed_formatter)
    error_handler.setLevel(logging.ERROR)
    
    # Console handler
    console_handler = logging.StreamHandler()
    console_handler.setFormatter(detailed_formatter)
    console_handler.setLevel(logging.INFO)
    
    # Root logger
    root_logger = logging.getLogger()
    root_logger.setLevel(getattr(logging, log_level))
    root_logger.addHandler(system_handler)
    root_logger.addHandler(error_handler)
    root_logger.addHandler(console_handler)
    
    return root_logger

import json
import logging

class StructuredLogger:
    """Logger that produces JSON-formatted logs"""
    
    def __init__(self, name):
        self.logger = logging.getLogger(name)
    
    def _log(self, level, message, **kwargs):
        log_entry = {
            'timestamp': datetime.now().isoformat(),
            'level': level,
            'logger': self.logger.name,
            'message': message,
            **kwargs
        }
        
        getattr(self.logger, level.lower())(json.dumps(log_entry))
    
    def info(self, message, **kwargs):
        self._log('INFO', message, **kwargs)
    
    def error(self, message, **kwargs):
        self._log('ERROR', message, **kwargs)
    
    def warning(self, message, **kwargs):
        self._log('WARNING', message, **kwargs)
    
    def debug(self, message, **kwargs):
        self._log('DEBUG', message, **kwargs)

# Usage

    """Extract metrics from log files"""
    
    for log_file in Path('data/logs').rglob('*.log'):
        # Update file size metric
        log_size_bytes.labels(file=str(log_file)).set(log_file.stat().st_size)
        
        # Count errors by level
        with open(log_file, 'r') as f:
            for line in f:
                if 'ERROR' in line:
                    log_errors_total.labels(level='ERROR').inc()
                elif 'WARNING' in line:
                    log_errors_total.labels(level='WARNING').inc()
                
                log_lines_total.labels(level='all').inc()

def recover_corrupted_log(log_file):
    """Attempt to recover corrupted log file"""
    
    recovered_lines = []
    corrupted_lines = []
    
    with open(log_file, 'r', errors='ignore') as f:
        for i, line in enumerate(f, 1):
            # Check if line has valid timestamp format
            if re.match(r'\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}', line):
                recovered_lines.append(line)
            else:
                corrupted_lines.append((i, line.strip()))
    
    # Save recovered logs
    recovered_file = log_file.with_suffix('.recovered.log')
    with open(recovered_file, 'w') as f:
        f.writelines(recovered_lines)
    
    return {
        'recovered': len(recovered_lines),
        'corrupted': len(corrupted_lines),
        'recovered_file': str(recovered_file)
    }

Do's and Don'ts
Do:

Include correlation IDs (request_id, user_id)

Log at appropriate levels

Use structured logging (JSON) in production

Rotate and compress old logs

Monitor log rates and sizes

Include context in error messages

Don't:

Log sensitive data (passwords, tokens, PII)

Log in tight loops (high-frequency logs)

Ignore log rotation

Mix different formats in the same file

Log without timestamps

Forget to handle log writing errors

Log Signing

import hashlib
import hmac

def sign_log_entry(entry, secret_key):
    """Sign log entry for tamper detection"""
    
    # Create HMAC signature
    signature = hmac.new(
        secret_key.encode(),
        entry.encode(),
        hashlib.sha256
    ).hexdigest()
    
    return f"{entry} | SIGNATURE={signature}"

def verify_log_entry(entry, secret_key):
    """Verify log entry signature"""
    
    if '| SIGNATURE=' not in entry:
        return False
    
    log_part, signature_part = entry.rsplit('| SIGNATURE=', 1)
    expected = hmac.new(
        secret_key.encode(),
        log_part.encode(),
        hashlib.sha256
    ).hexdigest()
    
    return signature_part.strip() == expected
